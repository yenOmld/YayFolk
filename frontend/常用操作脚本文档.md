# 常用操作脚本（含异常处理）文档

## 1. 脚本概述

本文档提供了前端和后端开发中常用的操作脚本示例，包含了完整的异常处理机制。这些脚本可以帮助开发者快速实现常见功能，同时确保代码的健壮性和可靠性。

### 1.1 适用场景

- 前端数据处理和API调用
- 后端服务操作和数据处理
- 自动化脚本和工具开发
- 数据迁移和批量处理

### 1.2 异常处理原则

1. **明确捕获**：只捕获预期的异常类型
2. **适当处理**：根据异常类型采取不同的处理策略
3. **详细记录**：记录异常信息以便调试
4. **优雅降级**：在异常发生时提供合理的备选方案
5. **用户友好**：向用户展示友好的错误信息

#### 2.1.1 基础 API 调用

```javascript
/**
 * 基础 API 调用函数
 * @param {string} url - API 地址
 * @param {Object} options - 请求选项
 * @returns {Promise} - 返回 Promise 对象
 */
async function apiCall(url, options = {}) {
  try {
    const defaultOptions = {
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${localStorage.getItem('token')}`
      },
      ...options
    };

    const response = await fetch(url, defaultOptions);

    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }

    const data = await response.json();

    if (data.code !== 200) {
      throw new Error(data.message || 'API error');
    }

    return data.data;
  } catch (error) {
    console.error('API call error:', error);
    // 可以在这里添加错误处理逻辑，如显示错误提示
    throw error;
  }
}

// 使用示例
async function getUserProfile() {
  try {
    const userProfile = await apiCall('/api/user/profile');
    console.log('User profile:', userProfile);
    return userProfile;
  } catch (error) {
    console.error('Failed to get user profile:', error);
    // 显示错误提示
    showError('获取用户信息失败，请稍后重试');
    return null;
  }
}
```

#### 2.1.2 带重试机制的 API 调用

```javascript
/**
 * 带重试机制的 API 调用
 * @param {string} url - API 地址
 * @param {Object} options - 请求选项
 * @param {number} retryCount - 重试次数
 * @param {number} retryDelay - 重试延迟（毫秒）
 * @returns {Promise} - 返回 Promise 对象
 */
async function apiCallWithRetry(url, options = {}, retryCount = 3, retryDelay = 1000) {
  let lastError;
  
  for (let i = 0; i < retryCount; i++) {
    try {
      return await apiCall(url, options);
    } catch (error) {
      lastError = error;
      // 只有网络错误才重试
      if (!error.message.includes('HTTP error')) {
        throw error;
      }
      
      console.log(`Retry ${i + 1}/${retryCount}...`);
      await new Promise(resolve => setTimeout(resolve, retryDelay));
    }
  }
  
  throw lastError;
}
```

### 2.2 表单处理脚本

```javascript
/**
 * 表单验证和提交
 * @param {HTMLFormElement} form - 表单元素
 * @param {Function} onSuccess - 成功回调
 * @param {Function} onError - 错误回调
 */
async function handleFormSubmit(form, onSuccess, onError) {
  try {
    // 表单验证
    if (!form.checkValidity()) {
      form.reportValidity();
      throw new Error('表单验证失败');
    }

    // 收集表单数据
    const formData = new FormData(form);
    const data = Object.fromEntries(formData.entries());

    // 显示加载状态
    const submitButton = form.querySelector('button[type="submit"]');
    const originalText = submitButton.textContent;
    submitButton.disabled = true;
    submitButton.textContent = '提交中...';

    // 提交数据
    const result = await apiCall(form.action, {
      method: form.method || 'POST',
      body: JSON.stringify(data)
    });

    // 恢复按钮状态
    submitButton.disabled = false;
    submitButton.textContent = originalText;

    // 调用成功回调
    if (onSuccess) {
      onSuccess(result);
    }
  } catch (error) {
    console.error('Form submission error:', error);
    
    // 恢复按钮状态
    const submitButton = form.querySelector('button[type="submit"]');
    if (submitButton) {
      submitButton.disabled = false;
      submitButton.textContent = '提交';
    }

    // 调用错误回调
    if (onError) {
      onError(error);
    } else {
      showError('提交失败，请稍后重试');
    }
  }
}

// 使用示例
const loginForm = document.getElementById('login-form');
loginForm.addEventListener('submit', (e) => {
  e.preventDefault();
  handleFormSubmit(
    loginForm,
    (result) => {
      showSuccess('登录成功');
      window.location.href = '/dashboard';
    },
    (error) => {
      showError('登录失败：' + (error.message || '请检查用户名和密码'));
    }
  );
});
```

```javascript
/**

// 带进度条的文件上传
function uploadFileWithProgress(file, url) {
  return new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest();
    const formData = new FormData();
    
    formData.append('file', file);
    
    xhr.upload.addEventListener('progress', (e) => {
      if (e.lengthComputable) {
        const percentComplete = Math.round((e.loaded / e.total) * 100);
        console.log(`Upload progress: ${percentComplete}%`);
        // 这里可以更新进度条
      }
    });
    
    xhr.addEventListener('load', () => {
      if (xhr.status === 200) {
        try {
          const data = JSON.parse(xhr.responseText);
          if (data.code === 200) {
            resolve(data.data);
          } else {
            reject(new Error(data.message || 'Upload failed'));
          }
        } catch (error) {
          reject(new Error('Invalid response format'));
        }
      } else {
        reject(new Error(`Upload failed: ${xhr.status}`));
      }
    });
    
    xhr.addEventListener('error', () => {
      reject(new Error('Network error during upload'));
    });
    
    xhr.open('POST', url);
    xhr.setRequestHeader('Authorization', `Bearer ${localStorage.getItem('token')}`);
    xhr.send(formData);
  });
}
```

```javascript
function createMySQLPool(config) {
  const pool = mysql.createPool({
    host: config.host,
    user: config.user,
    password: config.password,
    database: config.database,
    waitForConnections: true,
    connectionLimit: 10,
    queueLimit: 0
  });

  // 测试连接
  pool.getConnection()
    .then(connection => {
      console.log('MySQL connected successfully');
      connection.release();
    })
    .catch(error => {
      console.error('MySQL connection error:', error);
    });

  return pool;
}

/**
 * 执行 MySQL 查询
 * @param {Pool} pool - 连接池
 * @param {string} sql - SQL 语句
 * @param {Array} params - 参数数组
 * @returns {Promise} - 返回查询结果
 */
async function executeQuery(pool, sql, params = []) {
  let connection;
  try {
    connection = await pool.getConnection();
    const [rows, fields] = await connection.execute(sql, params);
    return rows;
  } catch (error) {
    console.error('MySQL query error:', error);
    throw error;
  } finally {
    if (connection) {
      connection.release();
    }
  }
}

/**
 * 执行事务
 * @param {Pool} pool - 连接池
 * @param {Function} callback - 事务回调
 * @returns {Promise} - 返回事务结果
 */
async function executeTransaction(pool, callback) {
  let connection;
  try {
    connection = await pool.getConnection();
    await connection.beginTransaction();
    
    const result = await callback(connection);
    
    await connection.commit();
    return result;
  } catch (error) {
    if (connection) {
      await connection.rollback();
    }
    console.error('Transaction error:', error);
    throw error;
  } finally {
    if (connection) {
      connection.release();
    }
  }
}
```

### 4.1 错误处理工具

```javascript
/**
 * 错误处理工具类
 */
class ErrorHandler {
  /**
   * 捕获并处理错误
   * @param {Function} fn - 要执行的函数
   * @param {Object} options - 选项
   * @returns {Promise} - 返回 Promise 对象
   */
  static async catchAsync(fn, options = {}) {
    try {
      return await fn();
    } catch (error) {
      this.handleError(error, options);
      return options.defaultValue;
    }
  }

  /**
   * 处理错误
   * @param {Error} error - 错误对象
   * @param {Object} options - 选项
   */
  static handleError(error, options = {}) {
    const { log = true, notify = false, message = '操作失败' } = options;

    if (log) {
      console.error(error);
    }

    if (notify) {
      // 这里可以集成通知系统
      console.log(`通知用户: ${message}`);
    }

    // 可以根据错误类型进行不同的处理
    if (error.code === 'ECONNREFUSED') {
      console.log('网络连接失败');
    } else if (error.code === 'ETIMEDOUT') {
      console.log('请求超时');
    }
  }

  /**
   * 重试函数
   * @param {Function} fn - 要执行的函数
   * @param {number} maxRetries - 最大重试次数
   * @param {number} delay - 重试延迟（毫秒）
   * @returns {Promise} - 返回 Promise 对象
   */
  static async retry(fn, maxRetries = 3, delay = 1000) {
    let lastError;

    for (let i = 0; i < maxRetries; i++) {
      try {
        return await fn();
      } catch (error) {
        lastError = error;
        console.log(`Retry ${i + 1}/${maxRetries}...`);
        await new Promise(resolve => setTimeout(resolve, delay));
      }
    }

    throw lastError;
  }
}

// 使用示例
async function fetchData() {
  return await ErrorHandler.catchAsync(async () => {
    const response = await fetch('/api/data');
    if (!response.ok) {
      throw new Error('Failed to fetch data');
    }
    return await response.json();
  }, {
    log: true,
    notify: true,
    defaultValue: null
  });
}
```

### 4.2 日志工具

```javascript
const fs = require('fs').promises;
const path = require('path');

/**
 * 日志工具类
 */
class Logger {
  constructor(logDir = './logs') {
    this.logDir = logDir;
    this.ensureLogDir();
  }

  /**
   * 确保日志目录存在
   */
  async ensureLogDir() {
    try {
      await fs.mkdir(this.logDir, { recursive: true });
    } catch (error) {
      console.error('Error creating log directory:', error);
    }
  }

  /**
   * 获取日志文件名
   * @returns {string} - 日志文件名
   */
  getLogFileName() {
    const today = new Date();
    const year = today.getFullYear();
    const month = String(today.getMonth() + 1).padStart(2, '0');
    const day = String(today.getDate()).padStart(2, '0');
    return `${year}-${month}-${day}.log`;
  }

  /**
   * 记录日志
   * @param {string} level - 日志级别
   * @param {string} message - 日志消息
   * @param {Object} data - 附加数据
   */
  async log(level, message, data = {}) {
    const timestamp = new Date().toISOString();
    const logEntry = {
      timestamp,
      level,
      message,
      data
    };

    console[level === 'error' ? 'error' : 'log'](`${timestamp} [${level.toUpperCase()}] ${message}`, data);

    try {
      const logFile = path.join(this.logDir, this.getLogFileName());
      await fs.appendFile(logFile, JSON.stringify(logEntry) + '\n');
    } catch (error) {
      console.error('Error writing to log file:', error);
    }
  }

  /**
   * 记录信息日志
   * @param {string} message - 日志消息
   * @param {Object} data - 附加数据
   */
  info(message, data = {}) {
    return this.log('info', message, data);
  }

  /**
   * 记录错误日志
   * @param {string} message - 日志消息
   * @param {Error} error - 错误对象
   */
  error(message, error) {
    return this.log('error', message, {
      error: {
        message: error.message,
        stack: error.stack
      }
    });
  }

  /**
   * 记录警告日志
   * @param {string} message - 日志消息
   * @param {Object} data - 附加数据
   */
  warn(message, data = {}) {
    return this.log('warn', message, data);
  }

  /**
   * 记录调试日志
   * @param {string} message - 日志消息
   * @param {Object} data - 附加数据
   */
  debug(message, data = {}) {
    return this.log('debug', message, data);
  }
}
```

### 5.1 前端数据同步脚本

```javascript
/**
 * 数据同步脚本
 * 用于同步本地数据和服务器数据
 */
class DataSync {
  constructor() {
    this.pendingSyncs = new Set();
  }

  /**
   * 同步数据
   * @param {string} endpoint - API 端点
   * @param {Object} data - 要同步的数据
   * @param {Function} onSuccess - 成功回调
   * @param {Function} onError - 错误回调
   */
  async syncData(endpoint, data, onSuccess, onError) {
    const syncId = `${endpoint}-${Date.now()}`;
    this.pendingSyncs.add(syncId);

    try {
      // 显示同步状态
      console.log(`开始同步数据到 ${endpoint}`);

      // 调用 API
      const result = await apiCall(endpoint, {
        method: 'POST',
        body: JSON.stringify(data)
      });

      // 移除待同步标记
      this.pendingSyncs.delete(syncId);

      // 调用成功回调
      if (onSuccess) {
        onSuccess(result);
      }

      console.log(`数据同步成功: ${endpoint}`);
      return result;
    } catch (error) {
      // 移除待同步标记
      this.pendingSyncs.delete(syncId);

      // 调用错误回调
      if (onError) {
        onError(error);
      } else {
        console.error(`数据同步失败: ${endpoint}`, error);
        showError('数据同步失败，请稍后重试');
      }

      throw error;
    }
  }

  /**
   * 批量同步数据
   * @param {Array} syncs - 同步任务数组
   * @returns {Promise} - 返回 Promise 对象
   */
  async batchSync(syncs) {
    const results = [];
    const errors = [];

    for (const sync of syncs) {
      try {
        const result = await this.syncData(
          sync.endpoint,
          sync.data,
          sync.onSuccess,
          sync.onError
        );
        results.push(result);
      } catch (error) {
        errors.push(error);
      }
    }

    return {
      results,
      errors,
      success: errors.length === 0
    };
  }

  /**
   * 检查是否有正在进行的同步
   * @returns {boolean} - 是否有正在进行的同步
   */
  hasPendingSyncs() {
    return this.pendingSyncs.size > 0;
  }

  /**
   * 取消所有正在进行的同步
   */
  cancelAllSyncs() {
    this.pendingSyncs.clear();
    console.log('所有同步任务已取消');
  }
}

// 使用示例
const dataSync = new DataSync();

async function syncUserProfile(profile) {
  try {
    await dataSync.syncData(
```

```javascript
const logger = require('./logger');
const { safeFind, safeUpdate } = require('./database');
const UserModel = require('../models/User');

/**
 * 用户数据处理服务
 */
class UserService {
  /**
   * 获取用户信息
   * @param {string} userId - 用户 ID
   * @returns {Promise} - 返回用户信息
   */
  async getUserById(userId) {
    try {
      logger.info(`获取用户信息: ${userId}`);
      
      const user = await safeFind(UserModel, { _id: userId });
      
      if (!user || user.length === 0) {
        throw new Error('用户不存在');
      }
      
      logger.info(`用户信息获取成功: ${userId}`);
      return user[0];
    } catch (error) {
      logger.error(`获取用户信息失败: ${userId}`, error);
      throw error;
    }
  }

  /**
   * 更新用户信息
   * @param {string} userId - 用户 ID
   * @param {Object} updates - 更新内容
   * @returns {Promise} - 返回更新结果
   */
  async updateUser(userId, updates) {
    try {
      logger.info(`更新用户信息: ${userId}`, updates);
      
      // 验证更新内容
      this.validateUserUpdate(updates);
      
      const result = await safeUpdate(
        UserModel,
        { _id: userId },
        { $set: updates },
        { new: true }
      );
      
      if (result.nModified === 0) {
        throw new Error('用户不存在或没有更新内容');
      }
      
      logger.info(`用户信息更新成功: ${userId}`);
      return result;
    } catch (error) {
      logger.error(`更新用户信息失败: ${userId}`, error);
      throw error;
    }
  }

  /**
   * 验证用户更新内容
   * @param {Object} updates - 更新内容
   */
  validateUserUpdate(updates) {
    // 检查不允许更新的字段
    const disallowedFields = ['_id', 'createdAt', 'role'];
    for (const field of disallowedFields) {
      if (updates[field]) {
        throw new Error(`不允许更新字段: ${field}`);
      }
    }

    // 验证邮箱格式
    if (updates.email) {
      const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
      if (!emailRegex.test(updates.email)) {
        throw new Error('邮箱格式不正确');
      }
    }

    // 验证密码长度
    if (updates.password) {
      if (updates.password.length < 6) {
        throw new Error('密码长度至少为 6 位');
      }
    }
  }

  /**
   * 批量更新用户状态
   * @param {Array} userIds - 用户 ID 数组
   * @param {Object} updates - 更新内容
   * @returns {Promise} - 返回更新结果
   */
  async batchUpdateUsers(userIds, updates) {
    try {
      logger.info(`批量更新用户状态`, { userIds, updates });
      
      const results = [];
      
      for (const userId of userIds) {
        try {
          const result = await this.updateUser(userId, updates);
          results.push({ userId, success: true, result });
        } catch (error) {
          results.push({ userId, success: false, error: error.message });
          logger.warn(`更新用户失败: ${userId}`, error);
        }
      }
      
      const successCount = results.filter(r => r.success).length;
      logger.info(`批量更新完成: ${successCount}/${userIds.length} 成功`);
      
      return results;
    } catch (error) {
      logger.error('批量更新用户失败', error);
      throw error;
    }
  }
}

// 导出服务实例
module.exports = new UserService();

// 使用示例

    console.error('用户更新失败:', error);
  }
}

async function batchUpdateUserStatus(userIds, status) {

    const results = await userService.batchUpdateUsers(userIds, { status });
    console.log('批量更新结果:', results);
  } catch (error) {
    console.error('批量更新失败:', error);
  }
}
```
